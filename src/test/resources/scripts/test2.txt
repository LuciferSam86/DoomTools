
check entry main(args)
{
	wad = wadfile(args[0]);
	mapview = mapview(wad, args[1]);
	close(wad);
	
	println("THINGS");
	each (i, t : mapview->things()) 
		println(i + ": " + t); 
	println("VERTICES");
	each (i, t : mapview->vertices()) 
		println(i + ": " + t); 
	println("LINEDEFS");
	each (i, t : mapview->linedefs()) 
		println(i + ": " + t); 
	println("SIDEDEFS");
	each (i, t : mapview->sidedefs()) 
		println(i + ": " + t); 
	println("SECTORS");
	each (i, t : mapview->sectors()) 
		println(i + ": " + t); 
}
