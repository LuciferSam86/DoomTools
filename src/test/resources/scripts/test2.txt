
check entry main(args)
{
	wad = wadfile(args[0]);
	doommap = map::view(wad, args[1]);
	close(wad);
	
	println("THINGS");
	each (i, t : doommap->map::things()) 
		println(i + ": " + t); 
	println("VERTICES");
	each (i, t : doommap->map::vertices()) 
		println(i + ": " + t); 
	println("LINEDEFS");
	each (i, t : doommap->map::linedefs()) 
		println(i + ": " + t); 
	println("SIDEDEFS");
	each (i, t : doommap->map::sidedefs()) 
		println(i + ": " + t); 
	println("SECTORS");
	each (i, t : doommap->map::sectors()) 
		println(i + ": " + t); 
}
